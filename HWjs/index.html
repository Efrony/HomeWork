1 . Дан код:

var a = 1, b = 1, c, d; - a++ или ++a это более каткая записи a = a + 1
c = ++a; alert(c); // 2 - префиксный декремент прибавляет 1 к текущему значению 'a' перед выполнением операции присваивания. 1+1=2
d = b++; alert(d); // 1 - постфиксный декремент. Сначала 'd' присваивается текущее значение 'b', а затем к 'b' прибаляется единица. т.е. d = b = 1
c = (2+ ++a); alert(c); // 5 - перед выполнением этой строки a = 2. Префиксный дикремент добаляет +1, т.е. а = 3 , следовательно, 3 + 2 = 5
d = (2+ b++); alert(d); // 4 - после выполнения второй строки b = 2. В данной строке 2 + 2 = 4. увеличение 'b' на 1 происходит после операции сложения и присваивания.
alert(a); // 3 - учитывая вышесказанное а = 2 + 1 = 3
alert(b); // 3 - аналогично, только 1 прибавилось к 'b' позже.
